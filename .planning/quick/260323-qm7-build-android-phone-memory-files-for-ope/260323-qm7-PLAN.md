class MemoryType(Enum):
    OS_GUIDE = "os"      # type: os
    APP_GUIDE = "app"    # type: app
    ICON_GUIDE = "icon"  # type: icon
    POLICY = "policy"    # type: policy

@dataclass(frozen=True)
class MemoryEntry:
    entry_id: str
    memory_type: MemoryType
    platform: str
    content: str
    app: str | None = None
    tags: tuple[str, ...] = ()
    created_at: float = field(default_factory=time.time)
    access_count: int = 0
```

# Entry serialization format (this is what the parser expects):
def _entry_to_section(entry: MemoryEntry) -> str:
    # Produces:
    #   ## {heading}
    #   id: {entry_id}
    #   type: {memory_type.value}     <-- "os", "app", "icon", or "policy"
    #   platform: {platform}
    #   app: {app_or_empty}
    #   tags: {comma-separated}
    #   created_at: {float_timestamp}
    #   access_count: {int}
    #
    #   {content text}

# Parser requires: id, type, platform all present, type must be valid enum value
# app: empty string = None, tags: empty = ()
# Blank line between metadata and content is REQUIRED
```
