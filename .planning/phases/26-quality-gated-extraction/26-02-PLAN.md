```python
class _AlwaysPassStepCritic:
    """Default step critic that passes every step."""
    async def evaluate(self, step: dict[str, Any], step_index: int) -> StepVerdict:
        return StepVerdict(step_index=step_index, passed=True, reason="always-pass default")

class _AlwaysPassTrajectoryCritic:
    """Default trajectory critic that passes every trajectory."""
    async def evaluate(self, steps: list[dict[str, Any]], metadata: dict[str, Any]) -> TrajectoryVerdict:
        return TrajectoryVerdict(passed=True, reason="always-pass default")


class ExtractionPipeline:
    """Orchestrate step-level and trajectory-level critics before skill production.

    Critic sequencing:
    1. Reject trajectories with fewer than 2 steps (no critic called).
    2. Run step_critic.evaluate() on each step in order. Stop at the first failure.
    3. Run trajectory_critic.evaluate() on the full step list + metadata.
    4. Call producer.produce() only when both critics pass.
    """

    def __init__(
        self,
        *,
        step_critic: StepCritic | None = None,
        trajectory_critic: TrajectoryCritic | None = None,
        producer: ShortcutSkillProducer | None = None,
    ) -> None:
        self._step_critic: StepCritic = step_critic or _AlwaysPassStepCritic()
        self._trajectory_critic: TrajectoryCritic = trajectory_critic or _AlwaysPassTrajectoryCritic()
        self._producer: ShortcutSkillProducer = producer or ShortcutSkillProducer()

    async def run(
        self,
        steps: list[dict[str, Any]],
        metadata: dict[str, Any],
    ) -> ExtractionSuccess | ExtractionRejected:
```

Pipeline `run()` implementation:
1. Check `len(steps) < 2` -> return `ExtractionRejected(reason="too_few_steps", failed_step_verdict=None, failed_trajectory_verdict=None)`
2. Loop over steps with enumerate: call `await self._step_critic.evaluate(step, i)` for each. Collect verdicts in a list. If any verdict has `passed=False`, immediately return `ExtractionRejected(reason="step_critic", failed_step_verdict=failed_verdict, failed_trajectory_verdict=None)`
3. Call `await self._trajectory_critic.evaluate(steps, metadata)`. If `passed=False`, return `ExtractionRejected(reason="trajectory_critic", failed_step_verdict=None, failed_trajectory_verdict=traj_verdict)`
4. Call `self._producer.produce(steps, app=metadata.get("app", "unknown"), platform=metadata.get("platform", "unknown"))`. Return `ExtractionSuccess(candidate=skill, step_verdicts=tuple(step_verdicts), trajectory_verdict=traj_verdict)`
