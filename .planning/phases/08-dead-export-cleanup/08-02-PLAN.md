3. Replace `_execute_and` with parallel implementation. Key design decisions:
   - Use `asyncio.Semaphore(self._max_concurrency)` to bound parallelism
   - Each child gets its own `child_completed: list[str]` to avoid shared-list mutation
   - After `asyncio.gather`, merge all child_completed lists into `context.completed`
   - If any child fails, trigger replan (same budget logic as before)
   - Collect all outputs and trace_paths from successful children

```python
async def _execute_and(self, node: Any, context: RouterContext) -> NodeResult:
    """AND: parallel execution with concurrency semaphore."""
    sem = asyncio.Semaphore(self._max_concurrency)
    all_outputs: list[str] = [""] * len(node.children)
    all_traces: list[list[str]] = [[] for _ in node.children]
    child_completed: list[list[str]] = [[] for _ in node.children]
    child_results: list[NodeResult | None] = [None] * len(node.children)

    async def _run_child(idx: int, child: Any) -> None:
        async with sem:
            result = await self.execute(child, RouterContext(
                task=context.task,
                completed=list(context.completed),  # snapshot, not shared
                gui_agent=context.gui_agent,
                tool_registry=context.tool_registry,
                mcp_client=context.mcp_client,
            ))
            child_results[idx] = result
            all_traces[idx] = result.trace_paths
            if result.success:
                all_outputs[idx] = result.output
                if child.node_type == "atom":
                    child_completed[idx].append(child.instruction)

    await asyncio.gather(*[_run_child(i, c) for i, c in enumerate(node.children)],
                         return_exceptions=False)

    # Merge completed instructions in order
    for cc in child_completed:
        context.completed.extend(cc)

    merged_traces = [t for traces in all_traces for t in traces]

    # Check for failures
    for i, result in enumerate(child_results):
        if result is not None and not result.success:
            # Attempt replan on first failure
            if self._planner is not None and self._replan_count < self._max_replans:
                replan_result = await self._try_replan_and(
                    node, i, node.children[i], context,
                    [o for o in all_outputs if o], merged_traces,
                )
                if replan_result is not None:
                    merged_traces.extend(replan_result.trace_paths)
                    if replan_result.success:
                        continue
            return NodeResult(
                success=False,
                output="\n".join(o for o in all_outputs if o),
                error=result.error or f"AND child {i} failed",
                trace_paths=merged_traces,
            )

    return NodeResult(
        success=True,
        output="\n".join(o for o in all_outputs if o),
        trace_paths=merged_traces,
    )
```

```python
async def _execute_or(self, node: Any, context: RouterContext) -> NodeResult:
    """OR: try children in priority order (mcp > tool > gui) until one succeeds."""
    all_traces: list[str] = []
    last_error: str | None = None

    # Sort children by capability priority — mcp > tool > gui
    sorted_children = sorted(
        node.children,
        key=lambda c: _CAPABILITY_PRIORITY.get(
            getattr(c, "capability", "gui"), 99
        ),
    )

    for child in sorted_children:
        result = await self.execute(child, context)
        all_traces.extend(result.trace_paths)
        if result.success:
            if child.node_type == "atom":
                context.completed.append(child.instruction)
            return NodeResult(success=True, output=result.output, trace_paths=all_traces)
        last_error = result.error

    # All children failed — attempt replan.
    if self._planner is not None and self._replan_count < self._max_replans:
        self._replan_count += 1
        try:
            new_plan = await self._planner.replan(
                context.task,
                completed=context.completed,
                failed="All OR alternatives failed",
                remaining=[],
            )
            replan_result = await self.execute(new_plan, context)
            all_traces.extend(replan_result.trace_paths)
            return replan_result
        except Exception as exc:
            logger.warning("Replanning after OR failure failed: %s", exc)

    return NodeResult(
        success=False,
        error=last_error or "All OR children failed",
        trace_paths=all_traces,
    )
```
