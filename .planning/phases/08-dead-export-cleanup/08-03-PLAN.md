2. Add `_needs_planning` method to `AgentLoop`:

```python
async def _needs_planning(self, task: str) -> bool:
    """One LLM call to assess whether task warrants decomposition."""
    messages = [
        {"role": "system", "content": (
            "You are a task complexity assessor. Determine if the user's task "
            "requires multi-step decomposition (multiple distinct capabilities, "
            "sequential multi-app operations, or parallel sub-tasks). "
            "Single-capability tasks with no sequential dependencies should return False."
        )},
        {"role": "user", "content": f"Task: {task}"},
    ]
    response = await self.provider.chat_with_retry(
        messages=messages,
        tools=[_COMPLEXITY_TOOL],
        model=self.model,
    )
    if response.tool_calls:
        args = response.tool_calls[0].arguments
        if isinstance(args, str):
            import json as _json
            try:
                args = _json.loads(args)
            except Exception:
                return False
        return bool(args.get("needs_planning", False))
    return False  # safe default
```

3. Add `_plan_and_execute` method to `AgentLoop`:

```python
async def _plan_and_execute(self, task: str) -> tuple[str | None, list[str], list[dict]]:
    """Plan the task via TaskPlanner and execute via TreeRouter."""
    from nanobot.agent.planner import TaskPlanner
    from nanobot.agent.router import RouterContext, TreeRouter

    planner = TaskPlanner(llm=self.provider)
    tree = await planner.plan(task)
    logger.info("Decomposed plan: %s", tree.to_dict())

    gui_tool = self.tools.get("gui_task")
    ctx = RouterContext(
        task=task,
        gui_agent=gui_tool,
        tool_registry=self.tools,
        mcp_client=self.tools,  # MCP tools also registered here
    )

    router = TreeRouter(planner=planner, max_replans=2)
    result = await router.execute(tree, ctx)
    output = result.output or ("Done." if result.success else result.error or "Task failed.")
    return output, ["task_planner", "tree_router"], []
```

NOTE: The TreeRouter's `_run_gui` calls `context.gui_agent.run(instruction, max_retries=1)` but `GuiSubagentTool` has `execute(task=...)` not `run()`. The router needs to be adapted. Two options:
- Override `_run_gui` in a subclass (cleanest)
- Wrap `GuiSubagentTool` in a thin adapter

Use a thin wrapper approach — add a local class in `_plan_and_execute`:

```python
class _GuiDispatchAdapter:
    """Adapts GuiSubagentTool.execute() to the interface TreeRouter._run_gui expects."""
    def __init__(self, tool):
        self._tool = tool
    async def run(self, instruction: str, max_retries: int = 1):
        import json as _json
        result_json = await self._tool.execute(task=instruction)
        data = _json.loads(result_json)
        from opengui.agent import AgentResult
        return AgentResult(
            success=data.get("success", False),
            summary=data.get("summary", ""),
            trace_path=data.get("trace_path"),
            steps_taken=data.get("steps_taken", 0),
            error=data.get("error"),
        )

gui_tool = self.tools.get("gui_task")
gui_adapter = _GuiDispatchAdapter(gui_tool) if gui_tool else None
ctx = RouterContext(task=task, gui_agent=gui_adapter, ...)
```
