    ```python
    class LLMConditionEvaluator:
        """Adapts :class:`~opengui.skills.executor.LLMStateValidator` to the
        :class:`ConditionEvaluator` protocol.

        ``LLMStateValidator`` exposes ``validate(valid_state: str, screenshot: Path) -> bool``
        while ``ConditionEvaluator`` requires ``evaluate(condition: StateDescriptor, screenshot: Path) -> bool``.
        This adapter maps ``StateDescriptor.value`` to ``valid_state`` and inverts the result
        when ``StateDescriptor.negated`` is ``True``.

        Use this adapter in both ``ShortcutApplicabilityRouter`` and ``ShortcutExecutor``
        instead of passing ``LLMStateValidator`` directly.
        """

        def __init__(self, state_validator: Any) -> None:
            self._validator = state_validator

        async def evaluate(self, condition: StateDescriptor, screenshot: Path) -> bool:
            result: bool = await self._validator.validate(condition.value, screenshot)
            return (not result) if condition.negated else result
    ```
