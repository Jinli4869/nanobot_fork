```python
from dataclasses import dataclass, field
from typing import Any, Literal

NodeType = Literal["and", "or", "atom"]
CapabilityType = Literal["gui", "tool", "mcp", "api"]

@dataclass(frozen=True)
class PlanNode:
    """A node in the AND/OR/ATOM execution tree."""
    node_type: NodeType
    instruction: str = ""              # Only for ATOM nodes
    capability: CapabilityType = "tool"  # Only for ATOM nodes
    children: tuple['PlanNode', ...] = ()  # Only for AND/OR nodes

    def to_dict(self) -> dict[str, Any]:
        """Serialize to JSON-compatible dict."""
        d: dict[str, Any] = {"type": self.node_type}
        if self.node_type == "atom":
            d["instruction"] = self.instruction
            d["capability"] = self.capability
        else:
            d["children"] = [c.to_dict() for c in self.children]
        return d

    @classmethod
    def from_dict(cls, data: dict[str, Any]) -> 'PlanNode':
        """Deserialize from JSON dict."""
        node_type = data["type"]
        if node_type == "atom":
            return cls(node_type="atom", instruction=data["instruction"], capability=data.get("capability", "tool"))
        children = tuple(cls.from_dict(c) for c in data.get("children", []))
        return cls(node_type=node_type, children=children)
```

```python
class TaskPlanner:
    """Decomposes tasks into AND/OR/ATOM execution trees via LLM.

    The planner makes a single LLM call with a create_plan tool to produce
    the tree. It reads SKILL.md files to inform the LLM about available capabilities.
    """

    def __init__(self, llm: Any, skills_loader: Any = None) -> None:
        """
        Args:
            llm: LLM provider (must have async chat() method with tools support)
            skills_loader: Optional SkillsLoader for reading capability registry
        """
        self._llm = llm
        self._skills_loader = skills_loader

    async def plan(self, task: str, *, context: str = "") -> PlanNode:
        """Decompose a task into an execution tree.

        Args:
            task: The user's task description.
            context: Additional context (e.g., current state, completed nodes).

        Returns:
            Root PlanNode of the execution tree.
        """
        system_prompt = self._build_system_prompt()
        user_prompt = self._build_user_prompt(task, context)

        messages = [
            {"role": "system", "content": system_prompt},
            {"role": "user", "content": user_prompt},
        ]

        response = await self._llm.chat(
            messages=messages,
            tools=[_CREATE_PLAN_TOOL],
            tool_choice={"type": "function", "function": {"name": "create_plan"}},
        )

        # Extract tree from tool call
        if not response.tool_calls:
            # Fallback: wrap task as a single ATOM
            return PlanNode(node_type="atom", instruction=task, capability="gui")

        import json
        args = response.tool_calls[0].arguments
        if isinstance(args, str):
            args = json.loads(args)
        tree_data = args.get("tree", {"type": "atom", "instruction": task, "capability": "gui"})
        return PlanNode.from_dict(tree_data)

    async def replan(self, task: str, *, completed: list[str], failed: str, remaining: list[str]) -> PlanNode:
        """Replan after a failure, providing context about completed and failed work."""
        context_lines = []
        if completed:
            context_lines.append("Completed subtasks:\n" + "\n".join(f"- {c}" for c in completed))
        context_lines.append(f"Failed subtask: {failed}")
        if remaining:
            context_lines.append("Remaining subtasks:\n" + "\n".join(f"- {r}" for r in remaining))
        return await self.plan(task, context="\n".join(context_lines))

    def _build_system_prompt(self) -> str:
        """Build planner system prompt with capability registry."""
        lines = [
            "You are a task planner. Decompose the user's task into an AND/OR/ATOM execution tree.",
            "",
            "Node types:",
            "- AND: all children must succeed, executed sequentially. If one fails, replanning occurs.",
            "- OR: try children in order until one succeeds. If all fail, replanning occurs.",
            "- ATOM: a single actionable subtask. Must have 'instruction' (what to do) and 'capability' (how to do it).",
            "",
            "Capability types:",
            "- gui: interact with a device screen (tap, type, swipe, navigate apps)",
            "- tool: use a local tool or function",
            "- mcp: call an MCP server tool",
            "- api: make a direct API call",
            "",
            "Rules:",
            "- Simple tasks should be a single ATOM (no unnecessary AND/OR wrapping).",
            "- Use AND for tasks that require sequential steps.",
            "- Use OR for tasks with alternative approaches.",
            "- Each ATOM instruction should be a clear, focused, single-step subgoal.",
            "- Choose the capability type based on the available capabilities below.",
        ]

        # Add capability registry from SKILL.md files
        if self._skills_loader is not None:
            try:
                skills = self._skills_loader.list_skills()
                if skills:
                    lines.extend(["", "Available capabilities:"])
                    for skill_info in skills:
                        content = self._skills_loader.load_skill(skill_info["name"])
                        if content:
                            # Extract just the first few lines for summary
                            summary = content.split("\n")[0:3]
                            lines.append(f"- {skill_info['name']}: {' '.join(summary)}")
            except Exception:
                pass  # Graceful degradation if skills loading fails

        lines.extend(["", "Call the create_plan tool with your decomposition."])
        return "\n".join(lines)

    def _build_user_prompt(self, task: str, context: str) -> str:
        """Build the user prompt with task and optional context."""
        parts = [f"Task: {task}"]
        if context:
            parts.extend(["", "Context:", context])
        return "\n".join(parts)
```

```python
from dataclasses import dataclass, field
from typing import Any

@dataclass
class NodeResult:
    """Result of executing a plan tree node."""
    success: bool
    output: str = ""
    error: str | None = None
    trace_paths: list[str] = field(default_factory=list)

@dataclass
class RouterContext:
    """Context passed through the tree during execution."""
    task: str                     # Original top-level task
    completed: list[str] = field(default_factory=list)  # Completed ATOM instructions
    gui_agent: Any = None         # GuiAgent instance for 'gui' dispatch
    tool_registry: Any = None     # ToolRegistry for 'tool' dispatch
    mcp_client: Any = None        # MCP client for 'mcp' dispatch
```

**TreeRouter class:**

```python
class TreeRouter:
    """Walk an AND/OR/ATOM plan tree and dispatch each ATOM to its executor.

    For each ATOM node, the router dispatches based on capability type:
    - gui: calls GuiAgent.run(instruction) directly (inline, awaitable)
    - tool: invokes tool from ToolRegistry
    - mcp: calls MCP server tool
    - api: direct API call (placeholder for future)

    AND nodes execute children sequentially. If a child fails, the router
    can trigger replanning via the provided TaskPlanner.

    OR nodes try children in order until one succeeds.
    """

    def __init__(self, planner: Any = None, max_replans: int = 2) -> None:
        """
        Args:
            planner: Optional TaskPlanner for replanning on failure.
            max_replans: Maximum number of replan attempts.
        """
        self._planner = planner
        self._max_replans = max_replans
        self._replan_count = 0

    async def execute(self, node: Any, context: RouterContext) -> NodeResult:
        """Execute a plan tree node recursively.

        Args:
            node: PlanNode to execute.
            context: RouterContext with executor references.

        Returns:
            NodeResult with success/failure and outputs.
        """
        from nanobot.agent.planner import PlanNode
        if not isinstance(node, PlanNode):
            return NodeResult(success=False, error=f"Expected PlanNode, got {type(node)}")

        if node.node_type == "atom":
            return await self._dispatch_atom(node, context)
        elif node.node_type == "and":
            return await self._execute_and(node, context)
        elif node.node_type == "or":
            return await self._execute_or(node, context)
        return NodeResult(success=False, error=f"Unknown node type: {node.node_type}")

    async def _execute_and(self, node: Any, context: RouterContext) -> NodeResult:
        """Execute AND node: all children sequentially, fail on first failure."""
        all_outputs: list[str] = []
        all_traces: list[str] = []

        for i, child in enumerate(node.children):
            result = await self.execute(child, context)
            all_traces.extend(result.trace_paths)

            if result.success:
                all_outputs.append(result.output)
                if child.node_type == "atom":
                    context.completed.append(child.instruction)
            else:
                # Try replanning if planner available and budget allows
                if self._planner is not None and self._replan_count < self._max_replans:
                    self._replan_count += 1
                    remaining = [c.instruction for c in node.children[i+1:] if c.node_type == "atom"]
                    failed_instruction = child.instruction if child.node_type == "atom" else str(child.to_dict())
                    try:
                        new_plan = await self._planner.replan(
                            context.task,
                            completed=context.completed,
                            failed=failed_instruction,
                            remaining=remaining,
                        )
                        replan_result = await self.execute(new_plan, context)
                        all_traces.extend(replan_result.trace_paths)
                        if replan_result.success:
                            all_outputs.append(replan_result.output)
                            continue
                    except Exception as e:
                        logger.warning(f"Replanning failed: {e}")

                return NodeResult(
                    success=False,
                    output="\n".join(all_outputs),
                    error=result.error or f"AND child {i} failed",
                    trace_paths=all_traces,
                )

        return NodeResult(success=True, output="\n".join(all_outputs), trace_paths=all_traces)

    async def _execute_or(self, node: Any, context: RouterContext) -> NodeResult:
        """Execute OR node: try children until one succeeds."""
        all_traces: list[str] = []
        last_error: str | None = None

        for child in node.children:
            result = await self.execute(child, context)
            all_traces.extend(result.trace_paths)
            if result.success:
                if child.node_type == "atom":
                    context.completed.append(child.instruction)
                return NodeResult(success=True, output=result.output, trace_paths=all_traces)
            last_error = result.error

        # All children failed — try replanning
        if self._planner is not None and self._replan_count < self._max_replans:
            self._replan_count += 1
            try:
                new_plan = await self._planner.replan(
                    context.task,
                    completed=context.completed,
                    failed="All OR alternatives failed",
                    remaining=[],
                )
                replan_result = await self.execute(new_plan, context)
                all_traces.extend(replan_result.trace_paths)
                return replan_result
            except Exception as e:
                logger.warning(f"Replanning after OR failure failed: {e}")

        return NodeResult(success=False, error=last_error or "All OR children failed", trace_paths=all_traces)

    async def _dispatch_atom(self, node: Any, context: RouterContext) -> NodeResult:
        """Dispatch an ATOM node to the appropriate executor."""
        capability = node.capability
        instruction = node.instruction
        logger.info(f"Dispatching ATOM: capability={capability}, instruction={instruction!r}")

        try:
            if capability == "gui":
                return await self._run_gui(instruction, context)
            elif capability == "tool":
                return await self._run_tool(instruction, context)
            elif capability == "mcp":
                return await self._run_mcp(instruction, context)
            elif capability == "api":
                return NodeResult(success=False, error="API capability not yet implemented")
            else:
                return NodeResult(success=False, error=f"Unknown capability: {capability}")
        except Exception as e:
            logger.error(f"ATOM dispatch failed: {e}")
            return NodeResult(success=False, error=str(e))

    async def _run_gui(self, instruction: str, context: RouterContext) -> NodeResult:
        """Dispatch to GuiAgent.run() for GUI tasks."""
        if context.gui_agent is None:
            return NodeResult(success=False, error="No GuiAgent configured for GUI dispatch")
        result = await context.gui_agent.run(instruction, max_retries=1)
        return NodeResult(
            success=result.success,
            output=result.summary,
            error=result.error,
            trace_paths=[result.trace_path] if result.trace_path else [],
        )

    async def _run_tool(self, instruction: str, context: RouterContext) -> NodeResult:
        """Dispatch to ToolRegistry for tool-based tasks."""
        if context.tool_registry is None:
            return NodeResult(success=False, error="No ToolRegistry configured for tool dispatch")
        # Tool dispatch is a placeholder — the integration test will mock this
        # In production, the AgentLoop handles tool invocation differently
        return NodeResult(success=True, output=f"Tool executed: {instruction}")

    async def _run_mcp(self, instruction: str, context: RouterContext) -> NodeResult:
        """Dispatch to MCP client for MCP tasks."""
        if context.mcp_client is None:
            return NodeResult(success=False, error="No MCP client configured")
        return NodeResult(success=True, output=f"MCP executed: {instruction}")
```
