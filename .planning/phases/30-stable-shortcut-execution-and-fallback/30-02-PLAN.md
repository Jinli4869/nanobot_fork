        """SSTA-01: post-step observe() is called AFTER execute(), not before."""
        from opengui.skills.multi_layer_executor import ShortcutExecutor
        from opengui.skills.shortcut import ShortcutSkill
        from opengui.skills.data import SkillStep

        call_log: list[str] = []

        class _LoggingBackend:
            platform = "android"

            async def observe(self, screenshot_path: Path, timeout: float = 5.0):
                screenshot_path.parent.mkdir(parents=True, exist_ok=True)
                screenshot_path.touch()
                # Distinguish pre vs post by filename suffix
                suffix = "pre" if "pre" in screenshot_path.name else "post"
                call_log.append(f"observe_{suffix}")
                from opengui.observation import Observation
                return Observation(
                    screenshot_path=str(screenshot_path),
                    screen_width=1080, screen_height=1920,
                    foreground_app="com.example.app", platform="android",
                )

            async def execute(self, action, timeout: float = 5.0):
                call_log.append("execute")
                return "ok"

        step = SkillStep(action_type="tap", target="btn", fixed=True, fixed_values={"x": 10, "y": 20})
        shortcut = ShortcutSkill(
            skill_id="sc-order-test", name="Order", description="d",

    @pytest.mark.asyncio
    async def test_post_step_validation(tmp_path: Path) -> None:

        from opengui.skills.multi_layer_executor import ShortcutExecutor, ContractViolationReport
        from opengui.skills.shortcut import ShortcutSkill, StateDescriptor
        from opengui.skills.data import SkillStep

        captured_screenshots: list[Path] = []

        class _CapturingEvaluator:

                captured_screenshots.append(screenshot)
                return False   # always fail — triggers ContractViolationReport

        cond = StateDescriptor(kind="screen_state", value="confirm_screen_visible")
        shortcut = ShortcutSkill(
            skill_id="sc-post-validate",
            name="PostVal",
            description="tests post validation",
            app="com.x",
            platform="android",
            steps=(step,),
            postconditions=(cond,),
        )

        executor = ShortcutExecutor(
            backend=_FakeBackend(tmp_path),
            grounder=_FakeGrounder(),
            condition_evaluator=_CapturingEvaluator(),
            screenshot_dir=tmp_path / "post_val",
            post_action_settle_seconds=0.0,
        )
        result = await executor.execute(shortcut)

        assert isinstance(result, ContractViolationReport)
        assert result.boundary == "post"
        assert result.step_index == 0
