```python
"""Adapter bridge: nanobot LLM/embedding providers -> opengui Protocol interfaces.

All adapter code lives in nanobot/ to keep opengui zero-dependency on nanobot.
"""
from __future__ import annotations

from typing import Any

import numpy as np

from nanobot.providers.base import LLMProvider as NanobotLLMProvider
from opengui.interfaces import LLMResponse as OpenguiLLMResponse, ToolCall


class NanobotLLMAdapter:
    """Wraps a nanobot LLMProvider to satisfy opengui's LLMProvider Protocol.

    Key mappings:
    - Delegates to chat_with_retry() for built-in retry on transient errors
    - Converts ToolCallRequest -> ToolCall (different frozen dataclass types)
    - Converts empty tool_calls list -> None (nanobot uses [], opengui uses None)
    - Converts content=None -> content="" (opengui LLMResponse.content is str, not Optional)
    - Drops reasoning_content, thinking_blocks, finish_reason, usage (opengui doesn't use them)
    - Stores full nanobot LLMResponse as raw for debugging
    """

    def __init__(self, provider: NanobotLLMProvider, model: str) -> None:
        self._provider = provider
        self._model = model

    async def chat(
        self,
        messages: list[dict[str, Any]],
        tools: list[dict[str, Any]] | None = None,
        tool_choice: str | None = None,
    ) -> OpenguiLLMResponse:
        kwargs: dict[str, Any] = {"messages": messages, "model": self._model}
        if tools is not None:
            kwargs["tools"] = tools
        if tool_choice is not None:
            kwargs["tool_choice"] = tool_choice

        nano_resp = await self._provider.chat_with_retry(**kwargs)

        opengui_tool_calls = [
            ToolCall(id=tc.id, name=tc.name, arguments=tc.arguments)
            for tc in nano_resp.tool_calls
        ] or None  # Convert empty list to None

        return OpenguiLLMResponse(
            content=nano_resp.content or "",
            tool_calls=opengui_tool_calls,
            raw=nano_resp,
        )
```

```python
class NanobotEmbeddingAdapter:
    """Wraps an async embedding callable to satisfy opengui's EmbeddingProvider Protocol.

    The embed_fn should be an async callable: (texts: list[str]) -> np.ndarray.
    If nanobot has no embedding API configured, do NOT instantiate this adapter —
    pass embedding_provider=None to GuiAgent (BM25-only retrieval still works).
    """

    def __init__(self, embed_fn) -> None:
        self._embed_fn = embed_fn

    async def embed(self, texts: list[str]) -> np.ndarray:
        return await self._embed_fn(texts)
```
